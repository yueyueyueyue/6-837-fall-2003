Using Stack Struct in C


void main()
{
    Stack S;

    Stack_Init(&S);

    Stack_Push(&S, 30);
    Stack_Push(&S, 40);

    printf("Top: %d\n", Stack_Top(&S));

    Stack_Pop(&S);
    printf("Top: %d\n", Stack_Top(&S));

    Stack_Pop(&S);
    printf("Top: %d\n", Stack_Top(&S));
}



Using Stack Class in C++


void main()
{
    Stack S;

    S.Push(30);
    S.Push(40);

    printf("Top: %d\n", S.Top());

    S.Pop();
    printf("Top: %d\n", S.Top());

    S.Pop();
    printf("Top: %d\n", S.Top());
}
